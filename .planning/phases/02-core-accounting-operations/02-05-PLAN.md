"""
Wave 5 — 4 facturacion handlers.

REGLAS INAMOVIBLES:
- VIN y Motor son OBLIGATORIOS en toda factura de venta de moto (FACT-01, FACT-03).
- Formato de item: "[Modelo] [Color] - VIN: [chasis] / Motor: [motor]" — exacto.
- Moto debe estar en estado "disponible" antes de facturar (FACT-03).
- Cascade post-factura: inventario_motos -> "Vendida", loanbook creado (FACT-02).
- inventario_motos y loanbook son datos OPERATIVOS — permitidos en MongoDB (CLAUDE.md).
- NUNCA escribir datos contables (montos, journals) en MongoDB.
"""
import datetime
from motor.motor_asyncio import AsyncIOMotorDatabase
from services.alegra.client import AlegraClient
from core.events import EventPublisher
from core.permissions import validate_write_permission


async def handle_crear_factura_venta_moto(
    tool_input: dict,
    alegra: AlegraClient,
    db: AsyncIOMotorDatabase,
    event_bus: EventPublisher,
    user_id: str,
) -> dict:
    """
    Crea factura de venta de moto en Alegra.
    VIN y Motor son campos OBLIGATORIOS — bloquear si faltan. (FACT-01, FACT-03)
    """
    validate_write_permission("contador", "POST /invoices", "alegra")

    vin = tool_input.get("vin", "").strip()
    motor = tool_input.get("motor", "").strip()
    modelo = tool_input.get("modelo", "")
    color = tool_input.get("color", "")
    precio_venta = tool_input["precio_venta"]
    cliente_alegra_id = tool_input["cliente_alegra_id"]
    item_alegra_id = tool_input.get("item_alegra_id")
    fecha = tool_input.get("fecha") or datetime.date.today().isoformat()

    # FACT-03: Blocking conditions
    if not vin:
        return {"success": False, "error": "VIN es obligatorio para facturar una moto. Proporcione el numero de chasis."}
    if not motor:
        return {"success": False, "error": "Motor es obligatorio para facturar una moto. Proporcione el numero de motor."}

    # Check moto status in inventario_motos
    moto = await db.inventario_motos.find_one({"vin": vin})
    if moto and moto.get("estado", "").lower() != "disponible":
        return {"success": False, "error": f"Moto VIN {vin} no disponible — estado actual: {moto.get('estado', 'desconocido')}"}

    # FACT-01: Mandatory item description format
    item_description = f"{modelo} {color} - VIN: {vin} / Motor: {motor}"

    invoice_payload = {
        "date": fecha,
        "client": {"id": cliente_alegra_id},
        "items": [
            {
                "id": item_alegra_id,
                "description": item_description,
                "quantity": 1,
                "price": precio_venta,
            }
        ],
        "observations": f"Venta moto VIN {vin}",
    }

    result = await alegra.request_with_verify("invoices", "POST", body=invoice_payload)
    factura_id = result["_alegra_id"]

    # FACT-02: Cascade — inventario_motos operativo
    await db.inventario_motos.update_one(
        {"vin": vin},
        {"$set": {"estado": "Vendida", "factura_alegra_id": factura_id}},
    )

    # FACT-02: Cascade — loanbook operativo
    loanbook_id = f"LB-{int(datetime.datetime.utcnow().timestamp())}"
    await db.loanbook.insert_one({
        "loanbook_id": loanbook_id,
        "vin": vin,
        "modelo": modelo,
        "color": color,
        "precio_venta": precio_venta,
        "cliente_alegra_id": cliente_alegra_id,
        "factura_alegra_id": factura_id,
        "estado": "pendiente_entrega",
        "created_at": datetime.datetime.utcnow(),
    })

    # FACT-02: Event
    await event_bus.publish(
        event_type="factura.venta.creada",
        source="agente_contador",
        datos={
            "alegra_id": factura_id,
            "vin": vin,
            "motor": motor,
            "modelo": modelo,
            "loanbook_id": loanbook_id,
            "precio_venta": precio_venta,
        },
    )

    return {
        "success": True,
        "alegra_id": factura_id,
        "loanbook_id": loanbook_id,
        "message": f"Factura #{factura_id} creada. Moto {modelo} VIN {vin} vendida. Loanbook {loanbook_id} creado en estado pendiente_entrega.",
    }


async def handle_consultar_facturas(
    tool_input: dict,
    alegra: AlegraClient,
    db: AsyncIOMotorDatabase,
    event_bus: EventPublisher,
    user_id: str,
) -> dict:
    """GET /invoices — read-only."""
    try:
        params = {}
        if tool_input.get("date_from"):
            params["date[from]"] = tool_input["date_from"]
        if tool_input.get("date_to"):
            params["date[to]"] = tool_input["date_to"]
        if tool_input.get("client_id"):
            params["client"] = tool_input["client_id"]
        data = await alegra.request_with_verify("invoices", "GET", params=params)
        return {"success": True, "data": data, "count": len(data) if isinstance(data, list) else 1}
    except Exception as e:
        return {"success": False, "error": f"Error consultando facturas: {str(e)}"}


async def handle_anular_factura(
    tool_input: dict,
    alegra: AlegraClient,
    db: AsyncIOMotorDatabase,
    event_bus: EventPublisher,
    user_id: str,
) -> dict:
    """Anula factura y revierte cascade: inventario -> Disponible, loanbook -> cancelado."""
    validate_write_permission("contador", "POST /invoices", "alegra")

    factura_id = tool_input["factura_id"]
    motivo = tool_input.get("motivo", "Anulacion sin motivo especificado")

    # Void invoice in Alegra
    result = await alegra.request_with_verify(f"invoices/{factura_id}/void", "POST", body={"observations": motivo})

    # Revert cascade
    await db.inventario_motos.update_one(
        {"factura_alegra_id": factura_id},
        {"$set": {"estado": "Disponible"}},
    )
    await db.loanbook.update_one(
        {"factura_alegra_id": factura_id},
        {"$set": {"estado": "cancelado", "motivo_cancelacion": motivo}},
    )

    await event_bus.publish(
        event_type="factura.venta.anulada",
        source="agente_contador",
        datos={"factura_id": factura_id, "motivo": motivo},
    )

    return {
        "success": True,
        "anulado_id": factura_id,
        "message": f"Factura #{factura_id} anulada. Inventario revertido a Disponible. Loanbook cancelado.",
    }


async def handle_crear_nota_credito(
    tool_input: dict,
    alegra: AlegraClient,
    db: AsyncIOMotorDatabase,
    event_bus: EventPublisher,
    user_id: str,
) -> dict:
    """POST /credit-notes."""
    validate_write_permission("contador", "POST /invoices", "alegra")

    payload = {
        "date": tool_input.get("fecha") or datetime.date.today().isoformat(),
        "invoiceId": tool_input["factura_id"],
        "observations": tool_input.get("motivo", "Nota credito"),
        "items": tool_input.get("items", []),
    }
    result = await alegra.request_with_verify("credit-notes", "POST", body=payload)

    await event_bus.publish(
        event_type="nota_credito.creada",
        source="agente_contador",
        datos={"alegra_id": result["_alegra_id"], "factura_id": tool_input["factura_id"]},
    )

    return {
        "success": True,
        "alegra_id": result["_alegra_id"],
        "message": f"Nota credito #{result['_alegra_id']} creada para factura {tool_input['factura_id']}.",
    }
