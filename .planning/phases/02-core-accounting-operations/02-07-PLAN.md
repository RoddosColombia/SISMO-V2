async def test_t2_retefuente_arriendo_entries_balance(dispatcher, mock_alegra):
    """T2: Gasto con ReteFuente 3.5% — entries balancean (sum debitos == sum creditos)."""
    captured_payload = {}

    async def capture_call(endpoint, method, body=None, params=None):
        if method == "POST" and endpoint == "journals":
            captured_payload.update(body or {})
        return {"_alegra_id": 999}

    mock_alegra.request_with_verify.side_effect = capture_call

    result = await dispatcher.dispatch(
        "registrar_gasto",
        {"descripcion": "arriendo bodega febrero", "monto": 1_000_000, "banco": "Bancolombia"},
        "user-1",
    )
    assert result["success"] is True
    entries = captured_payload.get("entries", [])
    total_debit = sum(e.get("debit", 0) for e in entries)
    total_credit = sum(e.get("credit", 0) for e in entries)
    assert round(total_debit, 2) == round(total_credit, 2), f"Entries unbalanced: debit={total_debit}, credit={total_credit}"

    """T3: Gasto de socio -> CXC, no gasto operativo."""
    captured_payload = {}

    async def capture_call(endpoint, method, body=None, params=None):
        if method == "POST":
            captured_payload.update(body or {})
        return {"_alegra_id": 888}

    result = await dispatcher.dispatch(
        "registrar_cxc_socio",

        "user-1",
    )
    assert result["success"] is True
    # Event should be cxc.socio.registrada, not gasto.causado
    publish_calls = mock_bus.publish.call_args_list
    event_types = [c.args[0] if c.args else c.kwargs.get("event_type") for c in publish_calls]
    assert any("cxc" in str(et) for et in event_types), f"Expected CXC event, got: {event_types}"
    # Entries should NOT debit a gasto account (5480, 5484, 5493, etc.)
    gasto_accounts = {5480, 5484, 5487, 5490, 5491, 5493, 5470, 5462}
    if "entries" in captured_payload:
